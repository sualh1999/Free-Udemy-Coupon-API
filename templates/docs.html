API Request Examples

curl -X GET \
-H "token: YOUR_TOKEN" \
https://couponhub.srachn.com/api/coupons/
import requests

headers = {
'token': 'YOUR_TOKEN'
}

response = requests.get(
'https://couponhub.srachn.com/api/coupons/',
headers=headers
)

print(response.json())
fetch('https://couponhub.srachn.com/api/coupons/', {
headers: {
  'token': 'YOUR_TOKEN'
}
})
.then(response => response.json())
.then(data => console.log(data));
const axios = require('axios');

axios.get('https://couponhub.srachn.com/api/coupons/', {
headers: {
  'token': 'YOUR_TOKEN'
}
})
.then(response => console.log(response.data));
